def euc_distance(a,b):
    return np.sqrt((a[0]- b[0])**2 + (a[1]-b[1])**2)

# Exact line search algorithm to decide on the steplength for the Cyclic Coordinate Search and Hook-Jeeves Method.
#f : function ; d ; direction ; x : kth value of x ; a and b : interval to search for best alpha ; eps : step length for incrementian 

def exact_line(f,d,x,a,b,eps):
    min_val = 1000000
    best_alpha = 0
    interval = np.arange(a,b,eps)
    for alpha in interval:
        if(f(x+(alpha*np.array(d)))<= min_val):
            min_val = f(x+(alpha*np.array(d)))
            best_alpha = alpha
    return best_alpha

alpha_vals = [] # array to keep alpha values at every iteration
x_vals = []  # array to keep x values at every iteration
x_next_vals = []  # array to keep x(k+1) values at every iteration
directions = []  # array to keep direction values at every iteration

# Cyclic search algortihm

def cyclic_search(eps1, f,a,b,eps2,x0):
    
    x = np.zeros((500,2))
    k = 0
    x[0] = x0 # initial points 
    I = np.identity(2) #Identity matrix to form directions for 2 dimension 
    
    while(True):
        for j in range(2):
            
            alpha = 0.
            alpha = exact_line(f,I[j],x[k],a,b,eps2)
            alpha_vals.append(alpha)
            directions.append(I[j])
            x[k+1][j] = x[k][j] + alpha
            x[k+1][1-j] = x[k][1-j] # line to go only one way 
            x_vals.append(x[k])
            x_next_vals.append([round(i,4) for i in x[k+1]])
            k= k+1
            
        if(euc_distance(x[k],x[k-1]) <= eps1): # euclidean distance as stopping condition 
            break
            
    return x[k] #optimum point

alpha_vals = [] # array to keep alpha values at every iteration
x_vals = [] # array to keep x values at every iteration
x_next_vals = [] # array to keep x(k+1) values at every iteration
x_temp_vals = [] # array to keep x(temp) values at every iteration
directions = []  # array to keep direction values at every iteration


#Hook-Jeeves Method function

def hook_jeeves_method(eps1, f,a,b,eps2,x0):
    
    x = np.zeros((500,2))
    d = np.zeros((500,2))
    Lambda = np.zeros(500)
    k = 0
    x[0] = x0
    I = np.identity(2)
    
    while(True):
        alphas = []
        x_temp = [[0.],[0.]] #array to keep x_temp values
        
        for j in range(2): # explotary move
            alpha = 0.
            alpha = exact_line(f,I[j],x[k],a,b,eps2)
            alphas.append(alpha)
            x_temp[j] = x[k][j] + alpha 
            
        alpha_vals.append([round(i,5) for i in alphas])
        x_temp_vals.append(np.round(x_temp, 4))
        
        d[k] = x_temp - x[k] # direction to from x(k) to x_temp
        
        directions.append(np.round(d[k],5))
        
        Lambda[k] = exact_line(f,d[k],x[k],a,b,eps2) # Lambda value optimization to go from x(k) to x(k+1)
        
        x[k+1] = x[k] +  Lambda[k] * d[k] # New x value 
        
        x_next_vals.append([round(i,4) for i in x[k+1]])
        x_vals.append(x[k])
        
        k= k+1
        
        if(euc_distance(x[k],x[k-1]) <= eps1): # Euclidean distance as stopping condition 
            break
    return x[k]

# Simplex search algorithm 

x_bar_vals = [] # array to keep x_bar values at every iteration
x_h_vals = [] # array to keep x_h values at every iteration
x_l_vals = [] # array to keep x_l values at every iteration
x_new_vals = [] # array to keep x(k+1) values at every iteration
Type = [] # array to keep the type of simplex 


def simplex_search(alpha ,beta ,gamma, eps, f , start_pts):
    
    x = np.zeros((3,2))
    x_h = np.array(start_pts[np.argmax([f(i) for i in start_pts])]) # x_h initialized 
    h = np.argmax([f(i) for i in start_pts]) # x_h index is saved 
    x_l = np.array(start_pts[np.argmin([f(i) for i in start_pts])]) # x_l initialized
    l = np.argmin([f(i) for i in start_pts]) # x_l index is saved
    
    # the other points which is going to be referred as x_m is initialized 
    
    for i in range(0,len(start_pts)):
        if((i != np.argmax([f(i) for i in x])) or (i !=np.argmin([f(i) for i in x]))):
            x_m = start_pts[i]
            m = i-1

    x = [start_pts[l],start_pts[m],start_pts[h]] # x_i values are formed i = 1,2,3
    
    
    while(True):
        
        #x_h and x_l values and their indexes are updated at every iteration with respect to their position 
        
        x_h = np.array(x[np.argmax([f(i) for i in x])])
        h = np.argmax([f(i) for i in x])
        x_l = np.array(x[np.argmin([f(i) for i in x])])
        l = np.argmin([f(i) for i in x])
        
        x_h_vals.append(x_h)
        x_l_vals.append(x_l)

        for i in range(0,len(start_pts)):
            if((i != np.argmax([f(i) for i in x])) or (i !=np.argmin([f(i) for i in x]))):
                x_m = x[i]
                m = i-1
         
        #block to calculate x_bar- summing the all x_i values and then take their average
        
        counter = 0
        Sum = np.zeros(2)
        for i in range(0,len(x)):
            if(np.array_equal(x[i], x_h) == False):
                counter +=1
                Sum += x[i]
        
        x_bar = Sum/counter
        x_bar = np.array(x_bar)
        
        x_bar_vals.append(x_bar)
        
        # Reflection
        Type_ = "R"
        
        x_r = np.array(x_bar) + alpha * (x_bar-x_h)
        x_r = np.array(x_r)
        
        if(f(x_l)>f(x_r)):
            
            #expansion
            Type_ = "E"
            x_e = x_bar + gamma * (x_r-x_bar)
            x_e = np.array(x_e)
            if(f(x_r)>f(x_e)):
                x_h = x_e
                x[h] = x_h
            else:
                Type_ = "R"
                x_h = x_r #expansion failed
                x[h] = x_h
        else:
            values = []
            for i in range(0,len(x)):
                
                if(np.array_equal(x[i], x_h) == False):
                    values.append(f(x[i]))
                    
            Max = np.max(values)
            
            if(Max >= f(x_r)):
                x_h = x_r
                x[h] = x_h
            
            else:
                Type_ = "C"
                #contraction
                if(f(x_h) > f(x_r)):
                    x_h_prime = x_r
                    x_h_prime = np.array(x_h_prime)
                else:
                    x_h_prime = x_h
                    x_h_prime = np.array(x_h_prime)
                x_c = x_bar + beta * (x_h_prime-x_bar)
                
                if(f(x_c)<= f(x_h_prime)):
                    x_h = x_c
                    x[h] = x_h
                else:
                    for i in range(0,len(x)):
                        x[i] = x[i] +  (x_l-x[i]) / 2
        
        x_new_vals.append(x_c)         
        Type.append(Type_)
        if(np.sqrt(1/(len(x) + 1) * np.sum([(f(i) - f(x_bar))**2 for i in x])) < eps): #stopping condition 
            break
        
    return x_bar # optimum x
